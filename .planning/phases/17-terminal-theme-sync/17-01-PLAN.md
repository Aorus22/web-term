<task type="auto">
  <name>Task 1: Create AppThemeProvider component</name>
  <files>fe/src/components/AppThemeProvider.tsx</files>
  <action>
    Create a new component `AppThemeProvider` that:
    1. Uses `useSettings()` to get the current `terminal_color_theme`.
    2. Uses `useTheme()` to get the current `theme_mode` and `resolvedTheme`.
    3. Maps the selected terminal theme colors (from `terminalThemes`) to CSS variables:
       - `--background`: theme.colors.background
       - `--foreground`: theme.colors.foreground
       - `--card`, `--popover`: theme.colors.background
       - `--primary`: theme.colors.blue
       - `--secondary`, `--muted`: theme.colors.black (or a darkened version of background)
       - `--accent`, `--border`: theme.colors.brightBlack
       - `--sidebar`, `--sidebar-border`: theme.colors.background, theme.colors.brightBlack
       - `--sidebar-foreground`: theme.colors.foreground
    4. Applies these variables to `document.documentElement.style` whenever the theme changes.
    5. Includes a utility to calculate luminance to set `--primary-foreground` (white/black) and to force `resolvedTheme` to 'light' or 'dark' if the terminal theme is explicitly light or dark (optional but recommended for cohesion).
  </action>
  <verify>
    <automated>Check if file exists and exports AppThemeProvider</automated>
  </verify>
  <done>AppThemeProvider created with mapping logic</done>
</task>
